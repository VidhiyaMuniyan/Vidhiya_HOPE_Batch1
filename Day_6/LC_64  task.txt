// 64. Minimum Path Sum

class Solution {
    public int minPathSum(int[][] g) {
        int m = g.length, n = g[0].length;

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {

                if (i == 0 && j == 0) continue;
                else if (i == 0) g[i][j] += g[i][j - 1];
                else if (j == 0) g[i][j] += g[i - 1][j];
                else g[i][j] += Math.min(g[i - 1][j], g[i][j - 1]);
            }
        }
        return g[m - 1][n - 1];
    }
}
