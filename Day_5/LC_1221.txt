// 1221. Split a String in Balanced Strings
class Solution {
    public int balancedStringSplit(String s) {
        int count = 0, result = 0;

        for (char c : s.toCharArray()) {
            if (c == 'R') count++;
            else count--;

            if (count == 0) result++;
        }

        return result;
    }
}
